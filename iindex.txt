function loadPlantsFromDB($pdo, $category = '', $difficulty = '', $search = '', $sortBy = 'name') {
    if (!$pdo) {

                'category' => '–°–∏—Å—Ç–µ–º–Ω–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ',
                'price' => 0,
                'main_image' => 'https://images.unsplash.com/photo-1544551763-46a013bb70d5?w=400',
                'description' => '–î–ª—è –ø—Ä–æ—Å–º–æ—Ç—Ä–∞ —Ç–æ–≤–∞—Ä–æ–≤ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ –∫ –±–∞–∑–µ –¥–∞–Ω–Ω—ã—Ö. –í—ã–ø–æ–ª–Ω–∏—Ç–µ –º–∏–≥—Ä–∞—Ü–∏—é –¥–∞–Ω–Ω—ã—Ö.',
                'care' => json_encode(['lighting' => '–ù/–î', 'temperature' => '–ù/–î', 'ph' => '–ù/–î']),
                'features' => json_encode(['–°–∏—Å—Ç–µ–º–Ω–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ']),
                'difficulty' => '–ù/–î',
                'status' => 'active'
            ]
        ];
    }

    try {
        // –°—Ç—Ä–æ–∏–º –∑–∞–ø—Ä–æ—Å —Å —Ñ–∏–ª—å—Ç—Ä–∞–º–∏
        $sql = "SELECT *, latin_name as latin FROM products WHERE status = 'active'";
        $params = [];

        if ($category) {
            $sql .= " AND category = ?";
            $params[] = $category;
        }

        if ($difficulty) {
            $sql .= " AND difficulty = ?";
            $params[] = $difficulty;
        }

        if ($search) {
            $sql .= " AND (name LIKE ? OR latin_name LIKE ? OR description LIKE ?)";
            $searchParam = "%{$search}%";
            $params[] = $searchParam;
            $params[] = $searchParam;
            $params[] = $searchParam;
        }

        // –î–æ–±–∞–≤–ª—è–µ–º —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫—É
        switch ($sortBy) {
            case 'price_asc':
                $sql .= " ORDER BY price ASC";
                break;
            case 'price_desc':
                $sql .= " ORDER BY price DESC";
                break;
            case 'difficulty':
                $sql .= " ORDER BY FIELD(difficulty, '–õ–µ–≥–∫–æ', '–°—Ä–µ–¥–Ω–µ', '–°–ª–æ–∂–Ω–æ', '–û—á–µ–Ω—å —Å–ª–æ–∂–Ω–æ', '–≠–∫—Å—Ç—Ä–µ–º–∞–ª—å–Ω–æ —Å–ª–æ–∂–Ω–æ')";
                break;

                $sql .= " ORDER BY name ASC";
        }

        $stmt = $pdo->prepare($sql);
        $stmt->execute($params);
        $plants = $stmt->fetchAll();

        // –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º JSON –ø–æ–ª—è –æ–±—Ä–∞—Ç–Ω–æ –≤ –º–∞—Å—Å–∏–≤—ã
        foreach ($plants as &$plant) {

            $plant['features'] = json_decode($plant['features'], true) ?: [];
            $plant['images'] = json_decode($plant['images'], true) ?: [];

            $plant['image'] = $plant['main_image'];
            $plant['latin'] = $plant['latin_name'];
        }

        return $plants;

    } catch (Exception $e) {
        error_log('Load plants error: ' . $e->getMessage());
        return [];
    }
}

            url.searchParams.delete('page'); // –°–±—Ä–∞—Å—ã–≤–∞–µ–º –ø–∞–≥–∏–Ω–∞—Ü–∏—é
            window.location.href = url.toString();
        }

        document.getElementById('searchInput')?.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                updateSearch();
            }
        });

        // –£–≤–µ–¥–æ–º–ª–µ–Ω–∏—è
        function showNotification(message, type = 'success') {
            const notification = document.createElement('div');
            notification.className = `alert alert-${type} alert-dismissible fade show position-fixed`;
            notification.style.cssText = 'top: 90px; right: 20px; z-index: 9999; max-width: 400px;';

            const icons = {
                'success': 'bi-check-circle',
                'error': 'bi-exclamation-triangle', 
                'warning': 'bi-exclamation-triangle',
                'info': 'bi-info-circle'
            };

            notification.innerHTML = `
                <div class="d-flex align-items-center">
                    <i class="${icons[type] || 'bi-info-circle'} me-2"></i>
                    ${message}
                </div>
                <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
            `;

            document.body.appendChild(notification);

            setTimeout(() => {
                if (notification.parentNode) {
                    notification.remove();
                }
            }, 5000);
        }

        // –ü–ª–∞–≤–Ω–∞—è –ø—Ä–æ–∫—Ä—É—Ç–∫–∞ –∫ —è–∫–æ—Ä—è–º
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            });
        });
