if (!function_exists('getAllProducts')) {
    require_once '../../functions.php';
}

// Проверяем, что функции загружены
if (!function_exists('getAllProducts')) {
    die('Ошибка: Не удалось загрузить функции');
}

// Обработка удаления товара
if (isset($_POST['delete_product'])) {
    $result = deleteProduct($_POST['product_id']);
    if ($result['success']) {
        echo '<div class="alert alert-success alert-dismissible fade show">
            <i class="fas fa-check-circle me-2"></i>Товар успешно удален!
            <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
        </div>';
    } else {
        echo '<div class="alert alert-danger alert-dismissible fade show">
            <i class="fas fa-exclamation-circle me-2"></i>Ошибка: ' . htmlspecialchars($result['error']) . '
            <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
        </div>';
    }
}

// Обработка копирования товара
if (isset($_POST['copy_product'])) {
    $result = copyProduct($_POST['product_id']);
    if ($result['success']) {
        echo '<div class="alert alert-success alert-dismissible fade show">
            <i class="fas fa-copy me-2"></i>Товар успешно скопирован! ID: ' . $result['product_id'] . '
            <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
        </div>';
    } else {
        echo '<div class="alert alert-danger alert-dismissible fade show">
            <i class="fas fa-exclamation-circle me-2"></i>Ошибка копирования: ' . htmlspecialchars($result['error']) . '
            <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
        </div>';
    }
}

// Обработка массовых действий
if (isset($_POST['bulk_action']) && !empty($_POST['selected_products'])) {
    $action = $_POST['bulk_action'];
    $productIds = $_POST['selected_products'];
    $results = [];

    switch ($action) {
        case 'delete':
            foreach ($productIds as $id) {
                $results[] = deleteProduct($id);
            }
            break;
        case 'activate':
            foreach ($productIds as $id) {
                $product = getProductById($id);
                if ($product) {
                    $product['status'] = 1;
                    $results[] = saveProduct($product);
                }
            }
            break;
        case 'deactivate':
            foreach ($productIds as $id) {
                $product = getProductById($id);
                if ($product) {
                    $product['status'] = 0;
                    $results[] = saveProduct($product);
                }
            }
            break;
    }

    $successCount = count(array_filter($results, function($r) { return $r['success']; }));
    echo '<div class="alert alert-info alert-dismissible fade show">
        <i class="fas fa-info-circle me-2"></i>Обработано товаров: ' . $successCount . ' из ' . count($productIds) . '
        <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
    </div>';
}

// Получение данных
$products = getAllProducts();
$categories = getCategories();
$search = $_GET['search'] ?? '';
$category_filter = $_GET['category'] ?? '';
$status_filter = $_GET['status'] ?? '';
$sort_by = $_GET['sort'] ?? 'name_asc';

// Фильтрация товаров
if ($search || $category_filter || $status_filter !== '') {
    $filtered_products = [];
    foreach ($products as $product) {
        $match = true;

        // Фильтр по поиску
        if ($search) {
            $searchLower = mb_strtolower($search);
            $nameMatch = mb_strpos(mb_strtolower($product['name']), $searchLower) !== false;
            $descMatch = mb_strpos(mb_strtolower($product['description'] ?? ''), $searchLower) !== false;
            $tagsMatch = mb_strpos(mb_strtolower($product['tags'] ?? ''), $searchLower) !== false;
            $skuMatch = mb_strpos(mb_strtolower($product['sku'] ?? ''), $searchLower) !== false;

            if (!($nameMatch || $descMatch || $tagsMatch || $skuMatch)) {
                $match = false;
            }
        }

        // Фильтр по категории
        if ($category_filter && $product['category_id'] != $category_filter) {
            $match = false;
        }

        // Фильтр по статусу
        if ($status_filter !== '' && ($product['status'] ?? 1) != $status_filter) {
            $match = false;
        }

        if ($match) {
            $filtered_products[] = $product;
        }
    }
    $products = $filtered_products;
}

// Сортировка товаров
usort($products, function($a, $b) use ($sort_by) {
    switch ($sort_by) {
        case 'name_desc':
            return strcasecmp($b['name'], $a['name']);
        case 'price_asc':
            return ($a['price'] ?? 0) <=> ($b['price'] ?? 0);
        case 'price_desc':
            return ($b['price'] ?? 0) <=> ($a['price'] ?? 0);
        case 'date_desc':
            return strcmp($b['created_at'] ?? '', $a['created_at'] ?? '');
        case 'date_asc':
            return strcmp($a['created_at'] ?? '', $b['created_at'] ?? '');
        default: // name_asc
            return strcasecmp($a['name'], $b['name']);
    }
});

// Статистика
$totalProducts = count(getAllProducts());
$activeProducts = count(array_filter($products, function($p) { return ($p['status'] ?? 1) == 1; }));
$inactiveProducts = $totalProducts - $activeProducts;
$totalValue = array_sum(array_map(function($p) { return ($p['price'] ?? 0) * ($p['stock_quantity'] ?? 0); }, $products));

function toggleSelectAll() {
    const selectAll = document.getElementById('selectAll');

        updateBulkActions();
        updateSelectAllState();
    }
});

// Обновление состояния главного чекбокса
function updateSelectAllState() {
    const checkboxes = document.querySelectorAll('.product-checkbox');
    const checkedBoxes = document.querySelectorAll('.product-checkbox:checked');
    const selectAll = document.getElementById('selectAll');

    if (selectAll) {
        if (checkedBoxes.length === 0) {
            selectAll.indeterminate = false;
            selectAll.checked = false;
        } else if (checkedBoxes.length === checkboxes.length) {
            selectAll.indeterminate = false;
            selectAll.checked = true;
        } else {
            selectAll.indeterminate = true;
            selectAll.checked = false;
        }
    }
}

// Обновление панели массовых действий
function updateBulkActions() {
    const checkedBoxes = document.querySelectorAll('.product-checkbox:checked');
    const bulkActions = document.getElementById('bulkActions');
    const selectedCount = document.getElementById('selectedCount');
    const selectedProducts = document.getElementById('selectedProducts');

    if (checkedBoxes.length > 0) {
        bulkActions.style.display = 'block';
        selectedCount.textContent = checkedBoxes.length;

        // Добавляем скрытые поля для выбранных товаров
        selectedProducts.innerHTML = '';
        checkedBoxes.forEach(checkbox => {
            const input = document.createElement('input');
            input.type = 'hidden';
            input.name = 'selected_products[]';
            input.value = checkbox.value;
            selectedProducts.appendChild(input);
        });
    } else {
        bulkActions.style.display = 'none';
    }
}

// Подтверждение массовых действий
function confirmBulkAction() {
    const action = document.querySelector('select[name="bulk_action"]').value;
    const count = document.querySelectorAll('.product-checkbox:checked').length;

    if (!action) {
        alert('⚠️ Выберите действие для выполнения');
        return false;
    }

    const messages = {
        'activate': `✅ Активировать ${count} товаров?`,
        'deactivate': `❌ Скрыть ${count} товаров?`,
        'delete': `🗑️ УДАЛИТЬ ${count} товаров?\n\n⚠️ Это действие НЕЛЬЗЯ отменить!`
    };

    return confirm(messages[action] || `Выполнить действие для ${count} товаров?`);
}

// Система уведомлений
function showNotification(message, type = 'info', duration = 3000) {
    const alertClass = {
        'success': 'alert-success',
        'error': 'alert-danger',
        'warning': 'alert-warning',
        'info': 'alert-info'
    }[type] || 'alert-info';

    const notification = document.createElement('div');
    notification.className = `alert ${alertClass} alert-dismissible fade show position-fixed`;
    notification.style.cssText = 'top: 20px; right: 20px; z-index: 9999; min-width: 300px;';
    notification.innerHTML = `
        ${message}
        <button type="button" class="btn-close" onclick="this.parentElement.remove()"></button>
    `;

    document.body.appendChild(notification);

    setTimeout(() => {
        if (notification.parentNode) {
            notification.remove();
        }
    }, duration);
}
